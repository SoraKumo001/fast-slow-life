# アイテムリスト：パラメータとデータ設計

ゲーム「Fast Slow Life」に登場する全アイテムのデータ定義およびパラメータ一覧です。Godotでの実装に向けたデータ構造も含みます。

---

## 1. アイテムのデータ構造 (GDScript)

アイテムデータはGodotの `Resource` を用いて一元管理できるように定義します。

```gdscript
# item_data.gd (イメージ)
class_name ItemData
extends Resource

enum Category {
	FOOD,                # 食料系
	WOOD,                # 木材系
	STONE,               # 石材系
	ORE,                 # 鉱石系
	MANA_STONE,          # 魔法石系
	MONSTER_MATERIAL,    # モンスター素材
	INTERMEDIATE,        # 中間素材（加工品）
	CONSUMABLE,          # 消費アイテム（薬品など）
	WEAPON,              # 武器（装備品）
	ARMOR                # 防具（装備品）
}

@export var id: String
@export var name: String
@export var category: Category
@export var description: String
@export var value_sell: int = 0  # 基本売却価格
@export var value_buy: int = 0   # 基本購入価格（交易所レベル1の時の価格）

# 装備・使用時の効果（Dictionary形式で定義）
# 例: {"hp_recover": 50}, {"str_boost": 10}
@export var effect_data: Dictionary = {}
```
